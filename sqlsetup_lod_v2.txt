
-- Add weight column to lod_sections
DO $$ 
BEGIN 
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='lod_sections' AND column_name='weight') THEN
    ALTER TABLE lod_sections ADD COLUMN weight NUMERIC DEFAULT 0;
  END IF;
END $$;

-- Create table for LOD Level Configurations (Ranges)
CREATE TABLE IF NOT EXISTS lod_level_configs (
    id SERIAL PRIMARY KEY,
    level INTEGER NOT NULL UNIQUE, -- 1 to 5
    min_score NUMERIC DEFAULT 0,
    max_score NUMERIC DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Insert default values if not exists
INSERT INTO lod_level_configs (level, min_score, max_score)
VALUES 
(1, 0, 20),
(2, 21, 40),
(3, 41, 60),
(4, 61, 80),
(5, 81, 100)
ON CONFLICT (level) DO NOTHING;
