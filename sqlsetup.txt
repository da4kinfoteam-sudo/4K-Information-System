-- Level of Development (LOD) Tables
CREATE TABLE IF NOT EXISTS lod_sections (
    id SERIAL PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    "order" INTEGER DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS lod_questions (
    id SERIAL PRIMARY KEY,
    section_id INTEGER REFERENCES lod_sections(id) ON DELETE CASCADE,
    text TEXT NOT NULL,
    weight NUMERIC DEFAULT 1.0,
    "order" INTEGER DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS lod_choices (
    id SERIAL PRIMARY KEY,
    question_id INTEGER REFERENCES lod_questions(id) ON DELETE CASCADE,
    text TEXT NOT NULL,
    points NUMERIC DEFAULT 0,
    "order" INTEGER DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS lod_assessments (
    id SERIAL PRIMARY KEY,
    ipo_id INTEGER REFERENCES ipos(id) ON DELETE CASCADE,
    year INTEGER NOT NULL,
    total_score NUMERIC DEFAULT 0,
    computed_level INTEGER DEFAULT 0,
    manual_level INTEGER,
    remarks TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    UNIQUE(ipo_id, year)
);

CREATE TABLE IF NOT EXISTS lod_answers (
    id SERIAL PRIMARY KEY,
    assessment_id INTEGER REFERENCES lod_assessments(id) ON DELETE CASCADE,
    question_id INTEGER REFERENCES lod_questions(id) ON DELETE CASCADE,
    choice_id INTEGER REFERENCES lod_choices(id) ON DELETE SET NULL,
    points_earned NUMERIC DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    UNIQUE(assessment_id, question_id)
);
