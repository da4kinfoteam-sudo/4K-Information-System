-- Add column to activities if not exists
DO $$ 
BEGIN 
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='activities' AND column_name='actualEndDate') THEN
    ALTER TABLE activities ADD COLUMN "actualEndDate" DATE;
  END IF;
END $$;
